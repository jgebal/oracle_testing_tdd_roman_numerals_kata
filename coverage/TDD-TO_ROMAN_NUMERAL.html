1 FUNCTION to_roman_numeral( pv_decimal_number INTEGER ) RETURN VARCHAR2 IS
 
2   lv_decimal_number INTEGER := pv_decimal_number;
 
3   lv_roman_numeral  VARCHAR2(100);
 
4 
5   TYPE la_decimal_roman_map IS TABLE OF decimal_to_roman;
 
6   lv_decimal_roman_map la_decimal_roman_map;
 
7 
8 BEGIN
 
9   IF lv_decimal_number > 3000 THEN
 
10     raise_application_error(-20000, 'Function works only up to 3000 decimal number');
 
11   END IF;
 
12   lv_decimal_roman_map
 
13     := la_decimal_roman_map(
 
14       decimal_to_roman(1000,'M'),
 
15       decimal_to_roman(900,'CM'), decimal_to_roman(500,'D'), decimal_to_roman(400,'CD'), decimal_to_roman(100,'C'),
 
16       decimal_to_roman(90,'XC'), decimal_to_roman(50,'L'), decimal_to_roman(40,'XL'), decimal_to_roman(10,'X'),
 
17       decimal_to_roman(9,'IX'), decimal_to_roman(5,'V'), decimal_to_roman(4,'IV'), decimal_to_roman(1,'I')
 
18       );
 
19   FOR i IN lv_decimal_roman_map.FIRST .. lv_decimal_roman_map.LAST LOOP
 
20     lv_roman_numeral := lv_roman_numeral || lv_decimal_roman_map(i).get(lv_decimal_number);
 
21   END LOOP;
 
22   RETURN lv_roman_numeral;
 
23 END;
